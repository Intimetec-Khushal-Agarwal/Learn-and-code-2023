class UserSignUpService
{

	public static void setupUser(UserDetail user) {
		boolean connectionStatus = DatabaseService.connectToDatabase(user)
		if(ConnectionStatus){
		   DatabaseService.createNewUser(user);
		   DatabaseService.storeDefaultData();
		   FileSystemManager.storeDirectory(user);
		   NotifyUser.notification(true);
		}
		else{
	   	   NotifyUser.notification(false);
	        }
	}
}
